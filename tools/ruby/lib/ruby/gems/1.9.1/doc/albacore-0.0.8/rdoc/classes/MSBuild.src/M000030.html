        def build_path_to_command
                win_dir = ENV['windir'] || ENV['WINDIR']
                win_dir = 'C:/Windows' if win_dir.nil?
                
                File.join(win_dir.dup, 'Microsoft.NET', 'Framework', 'v3.5', 'MSBuild.exe')
        end
