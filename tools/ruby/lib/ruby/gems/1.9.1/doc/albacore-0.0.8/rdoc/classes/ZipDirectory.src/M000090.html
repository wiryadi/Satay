        def zip_additional(zipfile)
                return if @additional_files.nil?
                @additional_files.reject{|f| reject_file(f)}.each do |file_path|
                        file_name = file_path.split('/').last if @flatten_zip
                        zipfile.add(file_name, file_path)
                end
        end
